get_ratio <- function(x, sex_cat) {
    return(filter(x, sex == sex_cat) %>% pull(ratio))
}

apply_test <- function(x, func, ...) {
    m <- get_ratio(x, "male") 
    f <- get_ratio(x, "female")
    return(tidy(func(f,m, ...)))    
}

apply_fc <- function(x) {
    len.samp <- 10e3
    m <- get_ratio(x, "male")
    f <- get_ratio(x, "female")
    mrep <- sample(m, len.samp, replace=TRUE)
    frep <- sample(f, len.samp, replace=TRUE)
    statistic <- median(frep/mrep)
    conf.int <- quantile(frep/mrep, prob=c(0.025, 0.975))
    tibble(statistic = statistic, conf.low = conf.int[1], conf.high=conf.int[2], method = "Fold-change by sampling")
}
