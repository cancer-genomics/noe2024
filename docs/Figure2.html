resample <- function(x, ...) {
    dots <- list(...)
    xpop <- nest(x, data=-unlist(dots))
    xsmp <- bind_cols(select(xpop, unlist(dots)), 
            tibble(data = lapply(xpop$data, function(x) x[sample.int(nrow(x), nrow(x), replace=TRUE),]))) 
    return(unnest(xsmp, data))                         
}

get_ratio <- function(x, sex_cat) {
    return(filter(x, sex == sex_cat) %>% pull(ratio))
}

apply_test <- function(x, func, ...) {
    m <- get_ratio(x, "male") 
    f <- get_ratio(x, "female")
    return(tidy(func(f,m, ...)))    
}

apply_fc <- function(x) {
    len.samp <- 1e3
    m <- get_ratio(x, "male")
    f <- get_ratio(x, "female")

    # compute median on resample 1000 times
    mrep_median <- apply(replicate(len.samp, sample(m, length(m), replace=TRUE)),2,median)
    frep_median <- apply(replicate(len.samp, sample(f, length(f), replace=TRUE)),2,median)

    # compute statistics
    fc <- frep_median/mrep_median
    statistic <- median(fc)
    conf.int <- quantile(fc, prob=c(0.025, 0.975))
    if(statistic >= 1) p.value <- mean(fc < 1) else p.value <- mean(fc >= 1)

    # return stat data
    tibble(statistic = statistic, conf.low = conf.int[1], conf.high=conf.int[2], p.value = p.value, method = "Fold-change by sampling", .fc_rep = list(fc))
}
